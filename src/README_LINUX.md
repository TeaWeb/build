# Commands
~~~bash
# start server
./bin/teaweb start

# stop server
./bin/teaweb stop

# restart server
./bin/teaweb restart

# print helps
./bin/teaweb -h

# print version
./bin/teaweb -v

# reload all proxy servers config
./bin/teaweb reload

# reset service state
./bin/teaweb reset

# print server status
./bin/teaweb status
~~~
